\begin{definition}\label{def:bool-syntax}
The language $\Lng_B(V)$ of \emph{boolean} formulas over a vocabulary $V$ is
given by the Backus--Naur form
  $\phi ::= \top \mid p \mid \lnot \phi \mid \phi \land \phi$
where $p \in V$.
\end{definition}
